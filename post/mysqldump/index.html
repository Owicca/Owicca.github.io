mysql:

  • to drop a table and ignore foreign key errors, use ALTER TABLE <name> DISABLE KEYS
  • when inserting multiple rows, if one row is invalid, no rows are inserted, to insert only valid ones and ignore errors from invalid ones use INSERT IGNORE INTO <table> ...

mysqldump:

  • --all-databases / -A
  • --databases / -B:
    • specified names are treated as db names
  • --tables / -T:
    • specified names are treated as table names, overrides -B
  • --no-create-db / -n:
    • don’t add CREATE DATABASE ... IF EXISTS
  • --no-data / -d:
    • don’t dump data
  • --no-create-info / -t:
    • don’t write table creation info
  • --add-drop-database
  • --add-drop-table
  • --add-drop-trigger
  • --add-locks
  • --skip-comments
  • --skip-lock-tables:
    • don’t read lock tables while dumping
  • --compact:
    • same as --skip-add-drop-table, --skip-add-locks, --skip-comments, --skip-disable-keys, --skip-set-charset
  • --compress / -C:
    • use compression in server/client protocol
  • --ignore-database=<name>:
    • dump every db except <name>
    • to ignore multiple dbs, simply use multiple times
  • --ignore-table-name=<name>:
    • dump every table except <name>
    • to ignore multiple tables, simply use multiple times
  • --ignore-table-data=<name>:
    • dump every table data except <name>
    • to ignore multiple table data, simply use multiple times
  • --lock-all-tables / -x:
    • lock all tables from all dbs during the dump
  • --quote-names / -Q:
    • quote table and column names with backtick
  • --replace:
    • use REPLACE INTO instead of INSERT INTO
